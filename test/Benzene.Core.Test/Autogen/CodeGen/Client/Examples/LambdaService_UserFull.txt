using System;
using System.Collections.Generic;
using System.Threading.Tasks;
using Benzene.Clients;
using Benzene.Clients.HealthChecks;
using Benzene.HealthChecks.Core;
using Benzene.Results;
using System.Diagnostics.CodeAnalysis;

namespace Benzene.Service.Clients.User
{
    [ExcludeFromCodeCoverage]
    public class UserServiceClient : IUserServiceClient
    {
        private readonly IBenzeneMessageSender _sender;

        public UserServiceClient(IBenzeneMessageSender sender)
        {
            _sender = sender;
        }

        public string HashCode => "819704d1f5276a20cd4cac80d9c75a2dd7f3ec96496380b5891692ecffdd346a";

        public Task<IBenzeneResult<UserDto>> GetUserAsync(GetUserMessage message)
        {
            return GetUserAsync(message, null);
        }

        public Task<IBenzeneResult<UserDto>> GetUserAsync(GetUserMessage message, IDictionary<string, string> headers)
        {
            return _sender.SendAsync<GetUserMessage, UserDto>("user:get", message, headers);
        }

        public Task<IBenzeneResult<string>> CreateUserAsync(CreateUserMessage message)
        {
            return CreateUserAsync(message, null);
        }

        public Task<IBenzeneResult<string>> CreateUserAsync(CreateUserMessage message, IDictionary<string, string> headers)
        {
            return _sender.SendAsync<CreateUserMessage, string>("user:create", message, headers);
        }

        public async Task<IBenzeneResult<HealthCheckResponse>> HealthCheckAsync()
        {
            var benzeneResult = await _sender.SendAsync<NullPayload, HealthCheckResponse>("healthcheck", new NullPayload());
            return benzeneResult.Status != BenzeneResultStatus.Ok
                ? benzeneResult
                : BenzeneResult.Ok(ClientHealthCheckProcessor.Process(benzeneResult.Payload, HashCode) as HealthCheckResponse);
        }
    }

    public static class UserServiceClientRouting
    {
        public static readonly string[] RequiredTopics = { "user:get", "user:create", "healthcheck" };
    }
}
