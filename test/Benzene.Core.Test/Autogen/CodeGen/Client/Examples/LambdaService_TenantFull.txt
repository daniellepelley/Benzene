using System;
using System.Collections.Generic;
using System.Threading.Tasks;
using Benzene.Clients;
using Benzene.Clients.HealthChecks;
using Benzene.HealthChecks.Core;
using Benzene.Results;
using System.Diagnostics.CodeAnalysis;

namespace Benzene.Service.Clients.Tenant
{
    [ExcludeFromCodeCoverage]
    public class TenantServiceClient : ITenantServiceClient
    {
        private readonly IBenzeneMessageSender _sender;

        public TenantServiceClient(IBenzeneMessageSender sender)
        {
            _sender = sender;
        }

        public string HashCode => "f7666334812315507767d314287501a0eb240dd2ebb58e6dc3731c9906eca6dd";

        public Task<IBenzeneResult<TenantDto>> GetTenantAsync(GetTenantMessage message)
        {
            return GetTenantAsync(message, null);
        }

        public Task<IBenzeneResult<TenantDto>> GetTenantAsync(GetTenantMessage message, IDictionary<string, string> headers)
        {
            return _sender.SendAsync<GetTenantMessage, TenantDto>("tenant:get", message, headers);
        }

        public Task<IBenzeneResult<TenantDto>> CreateTenantAsync(CreateTenantMessage message)
        {
            return CreateTenantAsync(message, null);
        }

        public Task<IBenzeneResult<TenantDto>> CreateTenantAsync(CreateTenantMessage message, IDictionary<string, string> headers)
        {
            return _sender.SendAsync<CreateTenantMessage, TenantDto>("tenant:create", message, headers);
        }

        public async Task<IBenzeneResult<HealthCheckResponse>> HealthCheckAsync()
        {
            var benzeneResult = await _sender.SendAsync<NullPayload, HealthCheckResponse>("healthcheck", new NullPayload());
            if (benzeneResult.Payload == null)
            {
                return benzeneResult;
            }
            var annotated = ClientHealthCheckProcessor.Process(benzeneResult.Payload, HashCode) as HealthCheckResponse;
            return BenzeneResult.Set(benzeneResult.Status, annotated, benzeneResult.IsSuccessful);
        }
    }

    public static class TenantServiceClientRouting
    {
        public static readonly string[] RequiredTopics = { "tenant:get", "tenant:create", "healthcheck" };
    }
}
